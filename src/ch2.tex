\chapter{Operations}
\label{ch:operations}
\vskip -4em \citep[chapter 2]{pinter_2016}

\begin{enumerate}[label={\Alph*.},font={\bfseries}]
  \item {\bf Examples of Operations}
    \begin{enumerate}[label={\arabic*},font={\bfseries}]
      \item $a \ast b = \sqrt{\abs{ab}}$ is not an operation on $\mathbb{Q}$,
        because e.g. $2 \ast 1 = \sqrt{\abs{2}} \not\in \mathbb{Q}$.
      \item $a \ast b = a \ln b$ is not an operation on $\mathbb{R}_{>0}$,
        because e.g. $\forall a, b \in \mathbb{R}_{>0} (b \le 1 \to a \ln b
        \not\in \mathbb{R}_{>0})$
      \item If $a \ast b$ is a root of the equation $x^{2} - a^{2}b^{2}=0$,
        $\ast$ is not an operation on $\mathbb{R}$, because
        $\forall a,b \in \mathbb{R} (a \ne 0 \land b \ne 0 \to x = \pm ab)$
      \item Subtraction is an operation on $\mathbb{Z}$, because
        $\forall a,b \in \mathbb{Z} (a-b \in \mathbb{Z})$.
      \item Subtraction is not an operation on $\mathbb{Z}_{\ge 0}$, because
        e.g. $0-1 \not\in \mathbb{Z}_{\ge 0}$.
      \item $a \ast b = \abs{a-b}$ is an operation on $\mathbb{Z}_{\ge 0}$,
        because $\forall a,b \in \mathbb{Z}_{\ge 0} (\abs{a-b} \in
        \mathbb{Z}_{\ge 0})$.
    \end{enumerate}
\end{enumerate}
